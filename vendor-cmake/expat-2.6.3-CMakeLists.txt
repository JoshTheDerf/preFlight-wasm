cmake_minimum_required(VERSION 3.10)
project(expat C)

# Minimal in-tree EXPAT build for the WASM target.
set(EXPAT_SRC_DIR ${CMAKE_CURRENT_SOURCE_DIR}/src/lib)

# Generate expat_config.h with the defines libexpat expects.
file(WRITE ${CMAKE_CURRENT_BINARY_DIR}/expat_config.h
"
#pragma once
/* No system entropy sources on wasm32. We define XML_POOR_ENTROPY so the
   parser uses a simple PRNG seeded from time(). The HAVE_* defines must be
   absent (not even 0) — expat checks with #ifdef, not #if. */
#define XML_POOR_ENTROPY 1
#define XML_GE 1
#define XML_DTD 1
#define XML_NS 1
#define XML_CONTEXT_BYTES 1024
#define BYTEORDER 1234
#define HAVE_MEMMOVE 1
#define HAVE_STDINT_H 1
#define HAVE_STDLIB_H 1
#define HAVE_STRING_H 1
#define HAVE_SYS_TYPES_H 1
#define HAVE_UNISTD_H 1
#define STDC_HEADERS 1
#define PACKAGE_BUGREPORT \"expat-bugs@libexpat.org\"
#define PACKAGE_NAME \"expat\"
#define PACKAGE_STRING \"expat 2.6.3\"
#define PACKAGE_TARNAME \"expat\"
#define PACKAGE_URL \"\"
#define PACKAGE_VERSION \"2.6.3\"
")

add_library(expat STATIC
    ${EXPAT_SRC_DIR}/xmlparse.c
    ${EXPAT_SRC_DIR}/xmlrole.c
    ${EXPAT_SRC_DIR}/xmltok.c
)

target_compile_definitions(expat PRIVATE
    HAVE_EXPAT_CONFIG_H=1
    XML_STATIC=1
)

target_include_directories(expat
    PUBLIC  ${EXPAT_SRC_DIR}
    PRIVATE ${CMAKE_CURRENT_BINARY_DIR}
)

# Aliases matching what libslic3r expects. We deliberately do NOT alias
# `libexpat` here because preflight's root CMakeLists creates its own
# INTERFACE target with that name and links EXPAT::EXPAT into it.
if (NOT TARGET EXPAT::EXPAT)
    add_library(EXPAT::EXPAT ALIAS expat)
endif ()
